create table warehouse(product_id varchar(10) primary key,
comp_name varchar(10),product_type varchar(10),purchase_date date,phone varchar(10));

create table products(product_id varchar(10) primary key,brand_name varchar(10),
product_type varchar(10) references warehouse(product_type),purchase_date date,expiry_date date,
foreign key(product_id) references warehouse(product_id) on delete cascade);

create table store(staff_id varchar(10),staff_name varchar(10),
staff_ph varchar(10),salary varchar(10),commission varchar(10),
no_of_orders varchar(10),primary key(staff_id,staff_ph));

create table customer(cust_id varchar(10) ,cust_name varchar(10),cust_ph varchar(10),
member_pts varchar(10),primary key(cust_id,cust_ph));

create table orders(order_id varchar(10) primary key,product_id varchar(10) references products(product_id),cust_name varchar(10) references customer(cust_name),
staff_id varchar(10),product_type varchar(10) references warehouse(product_type),purchase_amt varchar(10),
order_date date,
foreign key(staff_id) references store(staff_id));
